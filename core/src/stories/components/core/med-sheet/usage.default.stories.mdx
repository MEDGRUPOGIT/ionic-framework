## Typescript

```
import { Component } from '@angular/core';
import { BreakPointPixelEnum } from 'src/medsoft/app/shared/enums/breakpoint.enum';
import { BreakpointUtils } from 'src/medsoft/app/shared/utils/breakpoint.utils';
import {
  modalEnterAnimationSheetMobile,
  modalEnterAnimationSheetDesktop,
  modalLeaveAnimationSheetMobile,
  modalLeaveAnimationSheetDesktop
} from '../../../../shared/components/sheet-animation';

@Component({
  selector: 'component',
  templateUrl: './component.component.html',
  styleUrls: ['./component.component.scss']
})
export class Component {

  constructor(
    private readonly modalController: ModalController
  ) {}

  async abrirFiltros(): Promise<void> {
    const modal = await this.modalController.create({
      component: 'component',
      cssClass: 'tp-sheet',
      mode: 'ios',
      enterAnimation: BreakpointUtils.mediaQueryDown(BreakPointPixelEnum.LG)
        ? modalEnterAnimationSheetMobile
        : modalEnterAnimationSheetDesktop,
      leaveAnimation: BreakpointUtils.mediaQueryDown(BreakPointPixelEnum.LG)
        ? modalLeaveAnimationSheetMobile
        : modalLeaveAnimationSheetDesktop,
    });

    await modal.present();
  }
}
```
